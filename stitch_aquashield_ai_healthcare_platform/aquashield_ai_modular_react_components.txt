// --- MAIN PAGE ORCHESTRATOR ---

const App = () => {
  const [view, setView] = useState('landing'); // landing, loading, results
  const [loadingProgress, setLoadingProgress] = useState(0);
  const [loadingStep, setLoadingStep] = useState(0);

  const startAnalysis = () => {
    setView('loading');
    let p = 0;
    const interval = setInterval(() => {
      p += 1;
      setLoadingProgress(p);
      if (p === 20) setLoadingStep(1);
      if (p === 40) setLoadingStep(2);
      if (p === 60) setLoadingStep(3);
      if (p === 80) setLoadingStep(4);
      if (p === 100) {
        clearInterval(interval);
        setTimeout(() => setView('results'), 500);
      }
    }, 40);
  };

  return (
    <div className="font-sans antialiased text-slate-900 dark:text-slate-100 dark:bg-slate-950">
      <Navbar />
      
      {view === 'landing' && (
        <main className="animate-in fade-in duration-500">
          <Hero />
          <FeatureSection />
          <UploadComponent onUpload={startAnalysis} />
        </main>
      )}

      {view === 'loading' && (
        <LoadingScreen progress={loadingProgress} step={loadingStep} />
      )}

      {view === 'results' && (
        <main className="animate-in fade-in slide-in-from-bottom-8 duration-700">
          <Dashboard />
          <div className="max-w-7xl mx-auto px-6 pb-24 flex justify-center">
            <button 
              onClick={() => setView('landing')}
              className="text-slate-500 hover:text-blue-600 font-bold flex items-center gap-2 transition-colors"
            >
              ← Back to Upload
            </button>
          </div>
        </main>
      )}

      <Footer />
    </div>
  );
};

export default App;
